Interface Label

Defines a standard menu item with label & potentially an icon.

interface Label {
    condition?: boolean | () => boolean;
    icon?: string;
    imageAlt?: string;
    label: string;
    onPress?: (
        data?: {
            event?: PointerEvent | KeyboardEvent;
            focusSource?: A11yFocusSource;
            item?: Label;
        },
    ) => OnPressResult
    | Promise<OnPressResult>;
}
