Interface Stores

Defines all derived stores for positional properties. These property stores can be used to update the position state.

There are several readable stores for additional derived data.

interface Stores {
    dimension: Readable<
        {
            height: number
            | "auto"
            | "inherit";
            width: number | "auto" | "inherit";
        },
    >;
    element: Readable<HTMLElement>;
    height: WritablePos<string | number, number | "auto" | "inherit">;
    left: WritablePos<string | number, number>;
    maxHeight: WritablePos<string | number, number>;
    maxWidth: WritablePos<string | number, number>;
    minHeight: WritablePos<string | number, number>;
    minWidth: WritablePos<string | number, number>;
    resizeContentHeight: Readable<number>;
    resizeContentWidth: Readable<number>;
    resizeObservable: Readable<boolean>;
    resizeObserved: Writable<ResizeObject>;
    resizeOffsetHeight: Readable<number>;
    resizeOffsetWidth: Readable<number>;
    rotateX: WritablePos<string | number, number>;
    rotateY: WritablePos<string | number, number>;
    rotateZ: WritablePos<string | number, number>;
    scale: Writable<number>;
    top: WritablePos<string | number, number>;
    transform: Readable<TransformData>;
    transformOrigin: TransformOriginWritable;
    translateX: WritablePos<string | number, number>;
    translateY: WritablePos<string | number, number>;
    translateZ: WritablePos<string | number, number>;
    width: WritablePos<string | number, number | "auto" | "inherit">;
    zIndex: Writable<number>;
}
